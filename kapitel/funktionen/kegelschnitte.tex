\subsection{Kegelschnitte}

\subsubsection{Allgemeine Form}

\[
	A x^2 + B y^2 + C x + D y + E = \sigma
\]

\subsubsection{Kreis}

\[
	A = B
\]

\paragraph{Einheitskreis}

\[
	A = B = 1 \quad E = -1 \quad x^2 + y^2 = 1
\]

\paragraph{Hauptform vom Kreis}

\[
	{(x - x_0)}^2 + {(y - y_0)}^2 = r^2
\]


\subsubsection{Ellipse}

\[
	A \cdot B > 0, A \neq B
