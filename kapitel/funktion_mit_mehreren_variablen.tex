\section{Funktion mit mehreren Variablen}

\begin{align*}
	z & = f (x_1, x_2, x_3, \ldots)                                                                                                                                       \\
	z & = f (x, y)                                                                                                                                                        \\
	f & = \{ \lvert z (x, y) \rvert \quad (x, y) \in \underbrace{A \times B}_{\text{Definitionsbereich}},\quad z \in \underbrace{C}_{\text{Wertebereich}}, z = f(x, y) \}
\end{align*}

\paragraph{Beispiel}

\[
	z = x^2 + y^2 + 10
\]
